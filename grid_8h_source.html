123 dealii::Mapping<dim> const &
124 get_mapping(unsigned int const h_level, unsigned int const n_h_levels) const
125 {
126 if(mapping_dof_vector_fine_level.get())
127 {
128 // fine level
129 if(h_level == n_h_levels - 1)
130 {
131 return *(mapping_dof_vector_fine_level->get_mapping());
132 }
133 else // coarse levels
134 {
135 AssertThrow(h_level < mapping_dof_vector_coarse_levels.size(),
136 dealii::ExcMessage("Vector of coarse mappings seems to have incorrect size."));
137
138 return *(mapping_dof_vector_coarse_levels[h_level]->get_mapping());
139 }
140 }
141 else
142 {
143 AssertThrow(mapping_fine_level.get(),
144 dealii::ExcMessage("Fine-level mapping is uninitialized."));
145
146 return *mapping_fine_level;
147 }
148 }
