47template<int dim, int spacedim = dim>
48class BalancedGranularityPartitionPolicy
49 : public dealii::RepartitioningPolicyTools::Base<dim, spacedim>
50{
51public:
52 BalancedGranularityPartitionPolicy(unsigned int const n_mpi_processes)
53 : n_mpi_processes_per_level{n_mpi_processes}
54 {
55 }
56
57 virtual ~BalancedGranularityPartitionPolicy(){};
58
59 virtual dealii::LinearAlgebra::distributed::Vector<double>
60 partition(dealii::Triangulation<dim, spacedim> const & tria_coarse_in) const override
61 {
62 dealii::types::global_cell_index const n_cells = tria_coarse_in.n_global_active_cells();
63
64 // TODO: We hard-code a grain-size limit of 200 cells per processor
65 // (assuming linear finite elements and typical behavior of
66 // supercomputers). In case we have fewer cells on the fine level, we do
67 // not immediately go to 200 cells per rank, but limit the growth by a
68 // factor of 8, which limits makes sure that we do not create too many
69 // messages for individual MPI processes.
70 unsigned int const grain_size_limit =
71 std::min<unsigned int>(200, 8 * n_cells / n_mpi_processes_per_level.back() + 1);
72
73 dealii::RepartitioningPolicyTools::MinimalGranularityPolicy<dim, spacedim> partitioning_policy(
74 grain_size_limit);
75 dealii::LinearAlgebra::distributed::Vector<double> const partitions =
76 partitioning_policy.partition(tria_coarse_in);
77
78 // The vector 'partitions' contains the partition numbers. To get the
79 // number of partitions, we take the infinity norm.
80 n_mpi_processes_per_level.push_back(static_cast<unsigned int>(partitions.linfty_norm()) + 1);
81 return partitions;
82 }
83
84private:
85 mutable std::vector<unsigned int> n_mpi_processes_per_level;
86};

99 Grid(GridData const & data, MPI_Comm const & mpi_comm)
100 {
101 // triangulation
102 if(data.triangulation_type == TriangulationType::Serial)
103 {
104 AssertDimension(dealii::Utilities::MPI::n_mpi_processes(mpi_comm), 1);
105 triangulation = std::make_shared<dealii::Triangulation<dim>>();
106 }
107 else if(data.triangulation_type == TriangulationType::Distributed)
108 {
109 triangulation = std::make_shared<dealii::parallel::distributed::Triangulation<dim>>(
110 mpi_comm,
111 dealii::Triangulation<dim>::none,
112 dealii::parallel::distributed::Triangulation<dim>::construct_multigrid_hierarchy);
113 }
114 else if(data.triangulation_type == TriangulationType::FullyDistributed)
115 {
116 triangulation =
117 std::make_shared<dealii::parallel::fullydistributed::Triangulation<dim>>(mpi_comm);
118 }
119 else
120 {
121 AssertThrow(false, dealii::ExcMessage("Invalid parameter triangulation_type."));
122 }
123
124 // mapping
125 if(data.element_type == ElementType::Hypercube)
126 {
127 mapping = std::make_shared<dealii::MappingQ<dim>>(data.mapping_degree);
128 }
129 else if(data.element_type == ElementType::Simplex)
130 {
131 mapping =
132 std::make_shared<dealii::MappingFE<dim>>(dealii::FE_SimplexP<dim>(data.mapping_degree));
133 }
134 else
135 {
136 AssertThrow(false, dealii::ExcMessage("Invalid parameter element_type."));
137 }
138 }

221 void
222 do_create_triangulation(
223 GridData const & data,
224 std::function<void(dealii::Triangulation<dim> &)> const & create_triangulation,
225 unsigned int const global_refinements,
226 std::vector<unsigned int> const & vector_local_refinements)
227 {
228 if(data.triangulation_type == TriangulationType::Serial or
229 data.triangulation_type == TriangulationType::Distributed)
230 {
231 create_triangulation(*triangulation);
232
233 if(vector_local_refinements.size() > 0)
234 refine_local(*triangulation, vector_local_refinements);
235
236 if(global_refinements > 0)
237 triangulation->refine_global(global_refinements);
238 }
239 else if(data.triangulation_type == TriangulationType::FullyDistributed)
240 {
241 auto const serial_grid_generator = [&](dealii::Triangulation<dim, dim> & tria_serial) {
242 create_triangulation(tria_serial);
243
244 if(vector_local_refinements.size() > 0)
245 refine_local(tria_serial, vector_local_refinements);
246
247 if(global_refinements > 0)
248 tria_serial.refine_global(global_refinements);
249 };
250
251 auto const serial_grid_partitioner = [&](dealii::Triangulation<dim, dim> & tria_serial,
252 MPI_Comm const comm,
253 unsigned int const group_size) {
254 (void)group_size;
255 if(data.partitioning_type == PartitioningType::Metis)
256 {
257 dealii::GridTools::partition_triangulation(dealii::Utilities::MPI::n_mpi_processes(comm),
258 tria_serial);
259 }
260 else if(data.partitioning_type == PartitioningType::z_order)
261 {
262 dealii::GridTools::partition_triangulation_zorder(
263 dealii::Utilities::MPI::n_mpi_processes(comm), tria_serial);
264 }
265 else
266 {
267 AssertThrow(false, dealii::ExcNotImplemented());
268 }
269 };
270
271 unsigned int const group_size = 1;
272
273 // TODO SIMPLEX: this will not work in case of simplex meshes
274 auto const description = dealii::TriangulationDescription::Utilities::
275 create_description_from_triangulation_in_groups<dim, dim>(
276 serial_grid_generator,
277 serial_grid_partitioner,
278 triangulation->get_communicator(),
279 group_size,
280 dealii::Triangulation<dim>::none,
281 dealii::TriangulationDescription::construct_multigrid_hierarchy);
282
283 triangulation->create_triangulation(description);
284 }
285 else
286 {
287 AssertThrow(false, dealii::ExcMessage("Invalid parameter triangulation_type."));
288 }
289 }
